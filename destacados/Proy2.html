Sensor ultrasónico ESP32

Lectura de distancias usando el sensor HC-SR04 con ESP32.

Materiales

  • 1 ESP32
  • 1 Sensor ultrasónico HC-SR04
  • Jumpers
  • Protoboard

Conexiones

Trig a D5, Echo a D18, VCC a 5V, GND a GND.

Código


#define TRIG 5
#define ECHO 18

void setup() {
  Serial.begin(115200);
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);
}

void loop() {
  digitalWrite(TRIG, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG, LOW);

  long duracion = pulseIn(ECHO, HIGH);
  float distancia = duracion * 0.034 / 2;

  Serial.print(\"Distancia: \");
  Serial.print(distancia);
  Serial.println(\" cm\");
  delay(1000);
}
