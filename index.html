Robot sigue líneas con Arduino

Materiales:
  • Arduino UNO
  • 2 motores DC
  • 2 sensores IR TCRT5000
  • Driver L298N
  • Batería
  • Cables y protoboard
Conexiones: 
[Sensor IR izquierdo] → A0
[Sensor IR derecho] → A1
Motores ↔ Driver ↔ Pines 3, 4, 5, 6
Código: 
const int sensorIzq = A0;
const int sensorDer = A1;
const int motorIzqAdelante = 3;
const int motorIzqAtras = 4;
const int motorDerAdelante = 5;
const int motorDerAtras = 6;

void setup() {
  pinMode(sensorIzq, INPUT);
  pinMode(sensorDer, INPUT);
  pinMode(motorIzqAdelante, OUTPUT);
  pinMode(motorIzqAtras, OUTPUT);
  pinMode(motorDerAdelante, OUTPUT);
  pinMode(motorDerAtras, OUTPUT);
}

void loop() {
  int valIzq = analogRead(sensorIzq);
  int valDer = analogRead(sensorDer);
  int umbral = 500;

  if (valIzq < umbral && valDer < umbral) avanzar();
  else if (valIzq < umbral) girarIzq();
  else if (valDer < umbral) girarDer();
  else parar();
}

void avanzar() {
  digitalWrite(motorIzqAdelante, HIGH);
  digitalWrite(motorIzqAtras, LOW);
  digitalWrite(motorDerAdelante, HIGH);
  digitalWrite(motorDerAtras, LOW);
}

void girarIzq() {
  digitalWrite(motorIzqAdelante, LOW);
  digitalWrite(motorIzqAtras, HIGH);
  digitalWrite(motorDerAdelante, HIGH);
  digitalWrite(motorDerAtras, LOW);
}

void girarDer() {
  digitalWrite(motorIzqAdelante, HIGH);
  digitalWrite(motorIzqAtras, LOW);
  digitalWrite(motorDerAdelante, LOW);
  digitalWrite(motorDerAtras, HIGH);
}

void parar() {
  digitalWrite(motorIzqAdelante, LOW);
  digitalWrite(motorIzqAtras, LOW);
  digitalWrite(motorDerAdelante, LOW);
  digitalWrite(motorDerAtras, LOW);
}

Sensor ultrasónico con ESP32

Materiales:
  • ESP32
  • Sensor ultrasónico HC-SR04
  • Cables Dupont
  • Protoboard
Conexiones: 
VCC → 5V
GND → GND
TRIG → GPIO 5
ECHO → GPIO 18
Código: 
const int trigPin = 5;
const int echoPin = 18;

void setup() {
  Serial.begin(115200);
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH);
  float distance = duration * 0.034 / 2;

  Serial.print("Distancia: ");
  Serial.print(distance);
  Serial.println(" cm");

  delay(1000);
}

Control LED desde el móvil

Materiales:
  • ESP32
  • LED
  • Resistencia 220Ω
  • WiFi
Conexiones: 
LED al GPIO 2
Código: 
#include 
const char* ssid = "TU_SSID";
const char* password = "TU_PASSWORD";
WiFiServer server(80);

void setup() {
  pinMode(2, OUTPUT);
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(1000);
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (client) {
    String request = client.readStringUntil('\r');
    if (request.indexOf("LED=ON") >= 0) digitalWrite(2, HIGH);
    if (request.indexOf("LED=OFF") >= 0) digitalWrite(2, LOW);
    client.println("HTTP/1.1 200 OK\n\nLED Controlado");
    delay(1);
    client.stop();
  }
}

Monitor de temperatura WiFi

Materiales:
  • ESP32
  • Sensor de temperatura (ej. DHT11)
  • WiFi
Conexiones: 
DHT11 al GPIO 4
Código: 
#include 
#include 
DHT dht(4, DHT11);
const char* ssid = "TU_SSID";
const char* password = "TU_PASS";
WiFiServer server(80);

void setup() {
  Serial.begin(115200);
  dht.begin();
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(1000);
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (client) {
    float t = dht.readTemperature();
    client.println("HTTP/1.1 200 OK\n\nTemperatura: " + String(t) + " °C");
    delay(1);
    client.stop();
  }
}
